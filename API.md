# Librava API Documentation

Base URL:

http://librava.local/api

Authentication:
Authorization: Bearer YOUR_TOKEN

Endpoints:
GET    /books
GET    /books/{id}
POST   /books
PUT    /books/{id}
DELETE /books/{id}

GET    /borrows
GET    /borrows/{id}
POST   /borrows
POST   /borrows/{id}/return
POST   /borrows/{id}/renew
